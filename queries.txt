//Create role table
create table role (
	id serial PRIMARY KEY,
	rolename text not null,
	roletype INT 
)

//create user table
create table users (
	id serial PRIMARY KEY,
	name text NOT NULL,
	email text NOT NULL,
	password text NOT NULL,
	roleId int NOT NULL ,
	FOREIGN KEY (roleId) REFERENCES role (id)
)

//create menu table
create table menus (
	id serial PRIMARY KEY,
	menuname text not null 
)

//create sub menu table
create table submenus (
	id serial PRIMARY KEY,
	submenuname text not null,
	menuid int not null,
	FOREIGN KEY (menuid) REFERENCES menus (id)
)
